jQuery Plugin: vTree

Simple View


          $('#simple-list').vTree({url: 'data.json'});

Specific View


          $('#list').vTree({
            url: 'data.json',
            clickSelector: 'a', // This property sets selector for click event, which will be use to expand/collapse branch
            onLeafShow: function (el, data) { el.append('<a href="javascript:;">'+data.leaf_id+'</a> '+data.name); } // The leaf container will be filled by result of this callback (html)
          });

Catalog View


          $('#catalog').vTree({
            url: 'data.json',
            config: 'v-catalog',
            clickSelector: 'a', // This property sets selector for click event, which will be use to expand/collapse branch
            onLeafShow: function (el, data) { el.append('<a href="javascript:;">'+data.name+'</a>'); }
          });

Tree View


          $('#tree').vTree({
            url: 'data.json',
            config: 'v-tree',
            setLeaf: function () { // The default leaf container will be replaced by result of this callback (html)
              return $('<div style="border: 1px solid black; background: #eee; padding: 5px;"><a class="leaf-body" href="javascript:;"></a><div>');
            }
          });
