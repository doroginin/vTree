jQuery Plugin: Vega Tree

Simple View


      $('#simple-list').vTree({url: 'data.json'});

Specific View


      $('#list').vTree({
         url: 'data.json',
         clickSelector: 'a', // Задаем селектор для элементов по которым можно кликать, чтобы развернуть ветку
         onLeafShow: function (el, data) { el.append('<a href="javascript:;">'+data.leaf_id+'</a> '+data.name); } // Переопределяем Функцию, которая заполняет данными лист
      });

Catalog View


      $('#catalog').vTree({
         url: 'data.json',
         config: 'v-catalog',
         clickSelector: 'a', // Задаем селектор для элементов по которым можно кликать, чтобы развернуть ветку
         onLeafShow: function (el, data) { el.append('<a href="javascript:;">'+data.name+'</a>'); }
      });

Tree View


      $('#tree').vTree({
         url: 'data.json',
         config: 'v-tree',
         setLeaf: function () { // Переопределяем стандартный контейнер для листа, добавляем сслыку
            return $('<div style="border: 1px solid black; background: #eee; padding: 5px;"><a class="leaf-body" href="javascript:;"></a><div>');
         }
      });
